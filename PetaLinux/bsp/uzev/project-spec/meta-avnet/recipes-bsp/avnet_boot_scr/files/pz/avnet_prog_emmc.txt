# This is a boot script for U-Boot
# Generate boot.scr:
# mkimage -c none -A arm -T script -d boot.cmd.default boot.scr
#
################

# Commands to program emmc (image.ub and boot.scr) from USB

setenv copy_kernel_from_usb_to_emmc "echo Copying Kernel from USB to EMMC...; usb start && fatload usb 0:1 0x04000000 image_INITRD_FULL.ub && fatwrite mmc 1:1 '$fileaddr' image.ub '$filesize'"
setenv copy_boot_scr_from_usb_to_emmc "echo Copying Boot script from USB to EMMC...; usb start && fatload usb 0:1 0x04000000 avnet_mmc.scr && fatwrite mmc 1:1 '$fileaddr' boot.scr '$filesize'"
setenv erase_boot_scrip_qspi "echo Erasing QSPI Boot script...; sf probe 0 0 0 && sf erase '$script_offset_f' '$script_size_f'"

run copy_kernel_from_usb_to_emmc && run copy_boot_scr_from_usb_to_emmc && run erase_boot_scrip_qspi
